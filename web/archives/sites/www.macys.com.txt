}else if(pageType.match("^COMMUNITYPAGE")) {
redirectType = "Mblog";
}
}
cmExpAttributes[8] = redirectType;
}
/* Setting the 'US' country code as "1" or prefix with "1|" if already some value is there

var countryCode = cmExpAttributes[14];
